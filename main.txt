-- safe webhook
local function safeSendWebhook(text)
    if webhookUrl == nil or webhookUrl == "" then return end
    spawn(function()
        pcall(function()
            local data = {content = text}
            HttpService:PostAsync(webhookUrl, HttpService:JSONEncode(data), Enum.HttpContentType.ApplicationJson)
        end)
    end)
end

-- unlock logic
local function unlockUI()
    TweenService:Create(keyOverlay,TweenInfo.new(0.2),{BackgroundTransparency=1}):Play()
    wait(0.25)
    keyOverlay:Destroy()
    mainFrame.Visible = true
    safeSendWebhook("Correct key entered, UI unlocked.")
end

local function failKey()
    keyError.Text = "Invalid key"
    local orig = keyOverlay.Position
    TweenService:Create(keyOverlay, TweenInfo.new(0.06,Enum.EasingStyle.Linear,Enum.EasingDirection.Out,3,true), {Position=orig+UDim2.new(0,8,0,0)}):Play()
    delay(0.7,function() keyError.Text = "" end)
end

keyBtn.MouseButton1Click:Connect(function()
    if keyBox.Text == KEY then unlockUI() else failKey() end
end)

keyBox.FocusLost:Connect(function(enter)
    if enter then
        if keyBox.Text == KEY then unlockUI() else failKey() end
    end
end)

keyBox:CaptureFocus()
